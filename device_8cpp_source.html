47std::vector<std::string> evlist::InputDevice::partition(std::string str) {
48 if (str.empty()) {
49 return {};
50 }
51
52 std::ranges::transform(str, str.begin(), [](auto character) {
53 return std::tolower(character);
54 });
55
56 std::vector<std::string> result{};
57 result.emplace_back(std::string{str[0]});
58 for (std::size_t i = 1; i < str.length(); i++) {
59 auto prev = str[i - 1];
60 auto curr = str[i];
61 if (std::isdigit(prev) != 0 && std::isdigit(curr) != 0) {
62 result[result.size() - 1] += curr;
63 } else {
64 result.emplace_back(std::string{curr});
65 }
66 }
67
68 return result;
69}

88evlist::InputDevices& evlist::InputDevices::filter(
89 const std::vector<std::pair<Filter, std::string>>& filter, bool use_regex
90) {
91 auto capacity = devices_.size();
92 auto devices = std::move(devices_);
93 this->devices_ = {};
94 this->devices_.reserve(capacity);
95
96 for (auto device : devices) {
97 auto keep_device = std::ranges::all_of(
98 filter,
99 [&use_regex, &device, this](const auto& filter) {
100 if (use_regex) {
101 if (filter_regex(device, filter.first, filter.second)) {
102 return true;
103 }
104 } else {
105 if (filter_equality(device, filter.first, filter.second)) {
106 return true;
107 }
108 }
109
110 return false;
111 }
112 );
113
114 if (keep_device) {
115 this->devices_.emplace_back(std::move(device));
116 }
117 }
118
119 return *this;
120}

122bool evlist::InputDevices::filter_regex(
123 const InputDevice& device, Filter filter, const std::string& value
124) {
125 if (!regexes.contains(value)) {
126 regexes.emplace(value, std::regex{value});
