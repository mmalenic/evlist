289bool InputDevices::filter_device(
290 const InputDevice& device,
291 Filter filter,
292 std::invocable<const std::string&> auto comparison
293) {
294 switch (filter) {
295 case Filter::DEVICE_PATH:
296 return comparison(device.device_path().c_str());
297 case Filter::NAME:
298 return comparison(device.name());
299 case Filter::BY_ID:
300 return comparison(device.by_id().value_or(""));
301 case Filter::BY_PATH:
302 return comparison(device.by_path().value_or(""));
303 case Filter::CAPABILITIES:
304 const auto& capabilities = device.capabilities();
305 return std::ranges::any_of(
306 capabilities, [&comparison](const auto& capability) {
307 return comparison(capability);
308 }
309 );
310 }
311
312 return true;
313}

315/**
316 * Compare two input devices using natural sorting where multi-digit numbers
317 * are treated as a single number.
318 *
319 * @param lhs compare with left input device
320 * @param rhs compare with right input device
321 * @return the
322 * [`std::strong_ordering`](https://en.cppreference.com/w/cpp/utility/compare/strong_ordering)
323 * sort order
324 */
325inline auto operator<=>(const InputDevice& lhs, const InputDevice& rhs) {
326 auto lhs_device = lhs.device_path().string();
327 auto rhs_device = rhs.device_path().string();
328
329 auto s1_partitions = InputDevice::partition(lhs_device);
330 auto s2_partitions = InputDevice::partition(rhs_device);
331
332 for (auto [lhs, rhs] : std::views::zip(s1_partitions, s2_partitions)) {
333 if (lhs != rhs && !lhs.empty() && !rhs.empty()) {
334 if (std::isdigit(lhs[0]) != 0 && std::isdigit(rhs[0]) != 0) {
335 return std::stoi(lhs) <=> std::stoi(rhs);
336 }
337 return lhs <=> rhs;
338 }
339 }
340
341 return lhs_device <=> rhs_device;
342}

363 /**
364 * Format the input devices based on the output `Format`.
365 *
366 * @tparam Context context type
367 * @param devices input devices
368 * @param ctx context parameter
369 * @return iterator after formatting
370 */
371 template <typename Context>
372 // NOLINTNEXTLINE(runtime/references)
373 constexpr auto format(
374 const evlist::InputDevices& devices, Context& ctx
375 ) const {
376 auto csv_escape_quotes = [](std::string_view str) {
377 std::string out = {};
378 out.reserve(str.length());
379
380 for (const auto character : str) {
381 if (character == '"') {
382 out.push_back('"');
383 }
384 out.push_back(character);
385 }
386
387 return out;
388 };
389
390 auto format = [&csv_escape_quotes, &ctx, &devices](
391 std::string_view name,
392 std::string_view device,
393 std::string_view by_id,
394 std::string_view by_path,
395 std::string_view capabilities
396 ) {
397 switch (devices.output_format()) {
398 case evlist::Format::TABLE:
399 std::format_to(
400 ctx.out(),
401 "{:<{}}{: <{}}{: <{}}{: <{}}{}\n",
402 name,
403 devices.max_name(),
404 device,
405 devices.max_device_path(),
406 by_id,
407 devices.max_by_id(),
408 by_path,
409 devices.max_by_path(),
410 capabilities
411 );
412 break;
413 case evlist::Format::CSV:
414 std::format_to(
415 ctx.out(),
416 R"("{}","{}","{}","{}","{}")"
417 "\n",
418 csv_escape_quotes(name),
419 csv_escape_quotes(device),
420 csv_escape_quotes(by_id),
421 csv_escape_quotes(by_path),
422 csv_escape_quotes(capabilities)
423 );
424 break;
425 }
426 };
427
428 format(
429 evlist::InputDevices::HEADER_NAME,
430 evlist::InputDevices::HEADER_DEVICE_PATH,
431 evlist::InputDevices::HEADER_BY_ID,
432 evlist::InputDevices::HEADER_BY_PATH,
433 evlist::InputDevices::HEADER_CAPABILITIES
434 );
435
436 for (const auto& device : devices.devices()) {
437 auto capabilities = device.capabilities();
438 std::string capabilities_str{};
439 if (!capabilities.empty()) {
440 capabilities_str += "[";
441 for (auto name : device.capabilities()) {
442 capabilities_str += std::format("{}, ", name);
443 }
444
445 capabilities_str.erase(capabilities_str.length() - 2);
446 capabilities_str += "]";
447 }
448
449 format(
450 device.name(),
451 device.device_path().string(),
452 device.by_id().value_or(""),
453 device.by_path().value_or(""),
454 capabilities_str
455 );
456 }
457
458 return std::format_to(ctx.out(), "");
459 }
